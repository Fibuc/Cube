function flickerBackground() {
    let bg = document.querySelector("body::before");
    
    function flicker() {
        let randomOpacity = Math.random() * (1 - 0.8) + 0.8; // Entre 0.4 et 1
        document.body.style.setProperty('--bg-opacity', randomOpacity);
        
        setTimeout(flicker, Math.random() * (50 - 10) + 10); // Changement entre 100ms et 500ms
    }

    flicker();
}

document.addEventListener("DOMContentLoaded", flickerBackground);
